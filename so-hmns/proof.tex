\section{Theorem 1: High-Dimensional Manifold Smoothness}
Let $X \in \mathbb{R}^{N \times D}$ be the high-dimensional data and $L_{\text{sym}} = I - D^{-1/2}WD^{-1/2}$ be the Symmetrically Normalized Graph Laplacian.
\begin{theorem}
Minimizing $\text{Tr}(X^T L_{\text{sym}} X)$ guarantees geometric manifold smoothness.
\end{theorem}
\begin{proof}
The unnormalized graph Laplacian quadratic form evaluates as:
\begin{equation}
X^T L X = X^T (D - W) X = \frac{1}{2} \sum_{i,j=1}^N W_{ij} \Vert x_i - x_j \Vert_2^2
\end{equation}
Extending this to the symmetrically normalized matrix space $L_{\text{sym}}$, the trace operator yielding the Dirichlet energy evaluates to:
\begin{equation}
\text{Tr}(X^T L_{\text{sym}} X) = \frac{1}{2} \sum_{i,j=1}^N W_{ij} \left\Vert \frac{x_i}{\sqrt{D_{ii}}} - \frac{x_j}{\sqrt{D_{jj}}} \right\Vert_2^2
\end{equation}
As $\text{Tr}(X^T L_{\text{sym}} X) \to 0$, adjacent topological neighbors ($W_{ij} > 0$) are forced to minimize distance scales, preventing arbitrary manifold tearing.
\end{proof}
\section{Theorem 2: Eradication of System Singularity}
The dynamic learning rate is governed by the Boltzmann Attenuation Model: $\eta_m = \eta_0 \cdot \exp(-F_m / \kappa)$.
\begin{theorem}
The learning rate $\eta_m$ is strictly bounded, eliminating division-by-zero singularities.
\end{theorem}
\begin{proof}
Since free energy $F_m$ possesses a strict lower bound $F_{\text{min}} > -\infty$ for any finite discrete dataset, we bound the exponential scaling operator:
\begin{equation}
0 < \exp\left( -\frac{F_m}{\kappa} \right) \le \exp\left( -\frac{F_{\text{min}}}{\kappa} \right)
\end{equation}
Multiplying by the baseline velocity $\eta_0$ yields the bounded range:
\begin{equation}
\eta_m \in \left( 0, \, \eta_0 \exp\left( \frac{|F_{\text{min}}|}{\kappa} \right) \right]
\end{equation}
Therefore, $\eta_m$ never diverges to infinity, proving global numerical stability.
\end{proof}
\section{Theorem 3: Gradient-Norm Convergence}
Multi-objective loss utilizes dynamic weights updated via: $\lambda_i(t+1) = \lambda_i(t) + \alpha_{\text{grad}} (\mathcal{G}_i(t)/\bar{\mathcal{G}}(t) - 1)$.
\begin{theorem}
The dynamics of $\lambda_i(t)$ converge toward a balanced gradient norm equilibrium where $\mathcal{G}_i(t) = \bar{\mathcal{G}}(t)$.
\end{theorem}
\begin{proof}
We minimize the ratio error function: $L_{\text{grad}}(\bm{\lambda}) = \sum_{i} | \mathcal{G}_i(t)/\bar{\mathcal{G}}(t) - 1 |$. Taking the partial derivative with respect to $\lambda_i$:
\begin{equation}
\frac{\partial L_{\text{grad}}}{\partial \lambda_i} \propto \left( \frac{\mathcal{G}_i(t)}{\bar{\mathcal{G}}(t)} - 1 \right)
\end{equation}
If a task dominates ($\mathcal{G}_i(t) > \bar{\mathcal{G}}(t)$), the derivative is positive, causing the updater to subtract scale and decrease $\lambda_i(t+1)$. This negative feedback loop minimizes $L_{\text{grad}}$, proving convergence.
\end{proof}
